问题：实现响应式数据 + 依赖收集 ​

js
class Component {
data = { name: '' };
constructor() {}
render() {
console.log(`render - name:${this._data.name}`);
}
}

// 要求以下代码需要触发render，且同步变更需要合并。
const com = new Component();
com.data.name = 'a';
com.data.name = 'b';
com.data.name = 'c';
// 第一次触发 render

setTimeout(() => {
com.data.name = '修改';
});
// 第二次触发 render

答案：这种题在考察Vue nextTick原理,使用Object.defineProperty/Proxy实现响应式,使用微任务队列 实现render后置调用

js
class Component {
  data = {name: ''};
  _data = {name:''}

  pending = false

  constructor() {
    this.data = new Proxy(this.data, {
      set(target, key, value) {
        this._data[key] = value
        if(!this.pending){
          this.pending = true
          Promise.resolve().then(()=>{
            this.pending = false
            this.render()
          })
        }

      },
    })
  }

  render() {
    console.log(`render - name:${this._data.name}`);
  }
}


const com =  new Component()
com.data.name = 'a'
com.data.name = 'b'
com.data.name = 'c'
// 第一次触发 render

setTimeout(() => {
  com.data.name = '修改';
});
// 第二次触发 render
